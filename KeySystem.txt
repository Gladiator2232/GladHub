-- Advanced Script with Modern UI, Tween Animations, and Fade Effects

local correctKey = "GladBest" -- Correct key for script execution

local function createUI()
    -- ScreenGui setup
    local screenGui = Instance.new("ScreenGui")
    screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")

    -- Main Frame (with gradient background)
    local mainFrame = Instance.new("Frame")
    mainFrame.Parent = screenGui
    mainFrame.Size = UDim2.new(0, 350, 0, 350)
    mainFrame.Position = UDim2.new(0.5, -175, 0.5, -175)
    mainFrame.BackgroundColor3 = Color3.fromRGB(34, 34, 34)
    mainFrame.BackgroundTransparency = 0.5
    mainFrame.BorderSizePixel = 0

    -- Applying rounded corners to main frame
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 15)
    corner.Parent = mainFrame

    -- Tween for smooth appearance of the main frame
    mainFrame.Position = UDim2.new(0.5, -175, 0.5, -350)
    game:GetService("TweenService"):Create(mainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, -175, 0.5, -175)}):Play()

    -- Fade-in effect for UI elements
    game:GetService("TweenService"):Create(mainFrame, TweenInfo.new(1), {BackgroundTransparency = 0.3}):Play()

    -- Header Text (Stylish)
    local headerText = Instance.new("TextLabel")
    headerText.Parent = mainFrame
    headerText.Size = UDim2.new(1, 0, 0.2, 0)
    headerText.Position = UDim2.new(0, 0, 0, 0)
    headerText.Text = "Enter the key to unlock script:"
    headerText.TextColor3 = Color3.fromRGB(255, 255, 255)
    headerText.TextScaled = true
    headerText.BackgroundTransparency = 1
    headerText.Font = Enum.Font.GothamBold
    headerText.TextStrokeTransparency = 0.5
    headerText.TextStrokeColor3 = Color3.fromRGB(0, 255, 255)

    -- Fade-in for header
    game:GetService("TweenService"):Create(headerText, TweenInfo.new(1), {TextTransparency = 0}):Play()

    -- TextBox for Key Entry (With focus effects)
    local keyBox = Instance.new("TextBox")
    keyBox.Parent = mainFrame
    keyBox.Size = UDim2.new(0.8, 0, 0.2, 0)
    keyBox.Position = UDim2.new(0.1, 0, 0.25, 0)
    keyBox.PlaceholderText = "Insert your key here"
    keyBox.TextColor3 = Color3.fromRGB(255, 255, 255)
    keyBox.BackgroundColor3 = Color3.fromRGB(24, 24, 24)
    keyBox.BackgroundTransparency = 0.3
    keyBox.TextScaled = true
    keyBox.Font = Enum.Font.Gotham
    keyBox.Text = ""

    local boxCorner = Instance.new("UICorner")
    boxCorner.CornerRadius = UDim.new(0, 12)
    boxCorner.Parent = keyBox

    -- Fade-in for keyBox
    game:GetService("TweenService"):Create(keyBox, TweenInfo.new(1), {TextTransparency = 0}):Play()

    -- Tween for TextBox on focus
    keyBox.Focused:Connect(function()
        game:GetService("TweenService"):Create(keyBox, TweenInfo.new(0.3), {BackgroundTransparency = 0.1}):Play()
    end)
    keyBox.FocusLost:Connect(function()
        game:GetService("TweenService"):Create(keyBox, TweenInfo.new(0.3), {BackgroundTransparency = 0.3}):Play()
    end)

    -- Submit Button (with hover and click animations)
    local submitButton = Instance.new("TextButton")
    submitButton.Parent = mainFrame
    submitButton.Size = UDim2.new(0.8, 0, 0.2, 0)
    submitButton.Position = UDim2.new(0.1, 0, 0.6, 0)
    submitButton.Text = "Submit"
    submitButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    submitButton.TextScaled = true
    submitButton.BackgroundColor3 = Color3.fromRGB(34, 34, 34)
    submitButton.BackgroundTransparency = 0.5
    submitButton.Font = Enum.Font.GothamBold

    local submitButtonCorner = Instance.new("UICorner")
    submitButtonCorner.CornerRadius = UDim.new(0, 12)
    submitButtonCorner.Parent = submitButton

    -- Fade-in for Submit Button
    game:GetService("TweenService"):Create(submitButton, TweenInfo.new(1), {BackgroundTransparency = 0.5}):Play()

    -- Tween for Submit Button (hover effect)
    submitButton.MouseEnter:Connect(function()
        game:GetService("TweenService"):Create(submitButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(0, 255, 255)}):Play()
    end)

    submitButton.MouseLeave:Connect(function()
        game:GetService("TweenService"):Create(submitButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(34, 34, 34)}):Play()
    end)

    submitButton.MouseButton1Click:Connect(function()
        if keyBox.Text == correctKey then
            -- If correct, destroy UI and run the script
            game:GetService("TweenService"):Create(mainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, -175, 0.5, -350)}):Play()
            wait(0.5)
            screenGui:Destroy()
            loadstring(game:HttpGet("https://raw.githubusercontent.com/Gladiator2232/GladHub/refs/heads/main/GladHub.txt"))()
        else
            errorMessage.Text = "Incorrect key, try again."
            -- Fade-in error message
            game:GetService("TweenService"):Create(errorMessage, TweenInfo.new(0.5), {TextTransparency = 0}):Play()
        end
    end)

    -- Error message display (with fade-in and fade-out effects)
    local errorMessage = Instance.new("TextLabel")
    errorMessage.Parent = mainFrame
    errorMessage.Size = UDim2.new(1, 0, 0.2, 0)
    errorMessage.Position = UDim2.new(0, 0, 0.85, 0)
    errorMessage.Text = ""
    errorMessage.TextColor3 = Color3.fromRGB(255, 0, 0)
    errorMessage.TextScaled = true
    errorMessage.BackgroundTransparency = 1
    errorMessage.Font = Enum.Font.GothamBold
    errorMessage.TextTransparency = 1

    -- Fade-in the error message with a delay if incorrect
    errorMessage.TextTransparency = 1
    game:GetService("TweenService"):Create(errorMessage, TweenInfo.new(0.5), {TextTransparency = 0}):Play()

    -- Close Button (with hover effect)
    local closeButton = Instance.new("TextButton")
    closeButton.Parent = mainFrame
    closeButton.Size = UDim2.new(0.2, 0, 0.2, 0)
    closeButton.Position = UDim2.new(1, -40, 0, -40)
    closeButton.Text = "X"
    closeButton.TextColor3 = Color3.fromRGB(255, 255, 255)
    closeButton.TextScaled = true
    closeButton.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
    closeButton.BackgroundTransparency = 0.7
    closeButton.Font = Enum.Font.GothamBold

    local closeButtonCorner = Instance.new("UICorner")
    closeButtonCorner.CornerRadius = UDim.new(0, 12)
    closeButtonCorner.Parent = closeButton

    -- Fade-in for close button
    game:GetService("TweenService"):Create(closeButton, TweenInfo.new(1), {BackgroundTransparency = 0.7}):Play()

    closeButton.MouseEnter:Connect(function()
        game:GetService("TweenService"):Create(closeButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(255, 0, 0)}):Play()
    end)

    closeButton.MouseLeave:Connect(function()
        game:GetService("TweenService"):Create(closeButton, TweenInfo.new(0.2), {BackgroundColor3 = Color3.fromRGB(255, 0, 0)}):Play()
    end)

    closeButton.MouseButton1Click:Connect(function()
        -- Fade-out and destroy UI
        game:GetService("TweenService"):Create(mainFrame, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {Position = UDim2.new(0.5, -175, 0.5, -350)}):Play()
        wait(0.5)
        screenGui:Destroy()
    end)
end

-- Run the function to create the UI
createUI()
